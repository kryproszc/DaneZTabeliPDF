# Przykładowe dane
trojkat_laczny <- data.frame(
  `0 rok` = c(1000, 1200, 1100),
  `1 rok` = c(500, 600, 700),
  `2 lata` = c(300, 400, 500)
)

trojkat_zadoscuczynienia <- data.frame(
  `0 rok` = c(1000, 1200, 1100),
  `1 rok` = c(500, NA, NA),
  `2 lata` = c(NA, NA, NA)
)

# Obliczenie skumulowanych wartości
cumulative_laczny <- apply(trojkat_laczny, 2, cumsum)
cumulative_zadoscuczynienia <- apply(trojkat_zadoscuczynienia, 2, cumsum)

# Przekształcenie danych do formatu długiego do łatwiejszego wykresu
cumulative_data <- data.frame(
  Rok = rep(colnames(cumulative_laczny), each = nrow(cumulative_laczny)),
  Wartość = c(cumulative_laczny, cumulative_zadoscuczynienia),
  Typ = rep(c("Trójkąt Łączny", "Trójkąt Zadośćuczynienia"), times = ncol(cumulative_laczny))
)

# Usunięcie NA
cumulative_data <- na.omit(cumulative_data)

# Rysowanie wykresu
library(ggplot2)

ggplot(cumulative_data, aes(x = Rok, y = Wartość, color = Typ)) +
  geom_line() +
  geom_point() +
  labs(title = "Analiza Trendów: Trójkąt Łączny vs. Trójkąt Zadośćuczynienia",
       x = "Rok",
       y = "Skumulowana Wartość") +
  theme_minimal()
