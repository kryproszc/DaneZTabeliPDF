# Inicjalizacja pliku Excel
wb <- createWorkbook()

# Symulacja pętli z 4 iteracjami
for (i in 1:4) {
  # Tworzenie przykładowej tabeli danych
  tabela <- data.frame(
    ID = 1:5,
    Wartość = rnorm(5)
  )
  
  # Nazwa arkusza na podstawie iteracji
  sheet_name <- paste("Iteracja", i)
  
  # Dodanie nowego arkusza
  addWorksheet(wb, sheet_name)
  
  # Zapisanie tabeli do odpowiedniego arkusza
  writeData(wb, sheet_name, tabela)
}

# Zapisz plik Excel na dysku
saveWorkbook(wb, "wynik_symulacji.xlsx", overwrite = TRUE)
