import pandas as pd
import numpy as np

# Lista z liczbami i pustymi miejscami
data = [1, 2, np.nan, 4, 5]

# Tworzenie DataFrame z typem nullable integer
df = pd.DataFrame({'numbers': data}, dtype=pd.Int64Dtype())

# Dodanie 1 do każdej wartości w kolumnie 'numbers'
df['numbers'] = df['numbers'] + 1

# Sprawdzenie wyników
print(df)
print(df.dtypes)
