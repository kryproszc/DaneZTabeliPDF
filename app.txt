# Interactive console manipulation
repeat {
  cat("Podaj wartość i (lub 'exit' aby zakończyć): ")
  input_i <- readLines(con = stdin(), n = 1)
  if (tolower(input_i) == "exit") {
    cat("Kończę program.\n")
    break
  }
  i <- as.integer(input_i)
  
  cat("Podaj wartość j (lub 'exit' aby zakończyć): ")
  input_j <- readLines(con = stdin(), n = 1)
  if (tolower(input_j) == "exit") {
    cat("Kończę program.\n")
    break
  }
  j <- as.integer(input_j)
  
  if (is.na(i) || is.na(j)) {
    cat("Nieprawidłowe wartości. Proszę podać liczby całkowite.\n")
  } else {
    generate_plot(i, j)
  }
}
