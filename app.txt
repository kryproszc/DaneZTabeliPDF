library(dplyr)

# Funkcja do obliczenia kwantyla 0.99 dla każdego pliku CSV
process_csv_files <- function(folder_path) {
  # Lista plików CSV w folderze
  csv_files <- list.files(folder_path, pattern="*.csv", full.names=TRUE)
  
  # Inicjalizacja pustej listy do przechowywania wyników
  result_list <- list()
  
  # Przetwarzanie każdego pliku CSV
  for (csv_file in csv_files) {
    # Wczytanie danych z pliku CSV
    data <- read.csv(csv_file)
    
    # Obliczenie kwantyla 0.99 dla każdej kolumny
    quantiles <- sapply(data, function(column) quantile(column, probs=0.99, na.rm=TRUE))
    
    # Utworzenie data frame z nazwą pliku i kwantylami
    df <- data.frame(File=basename(csv_file), t(quantiles))
    
    # Dodanie wyników do listy
    result_list[[basename(csv_file)]] <- df
  }
  
  # Połączenie wszystkich wyników w jeden data frame
  final_df <- bind_rows(result_list)
  return(final_df)
}

# Użycie funkcji
folder_path <- "path/to/your/folder" # Zastąp swoją ścieżką do folderu
summary_df <- process_csv_files(folder_path)

# Wyświetlenie wyników
print(summary_df)
