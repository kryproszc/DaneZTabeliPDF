# Grupowanie po Szerokosc i Dlugosc i liczenie wystąpień każdej pary
powtorzenia <- aggregate(cbind(Szerokosc, Dlugosc) ~ Szerokosc + Dlugosc, data = df, length)

# Wypisywanie par, które się powtarzają (liczba wystąpień > 1)
powtarzajace_sie <- powtorzenia[powtorzenia$Szerokosc > 1, ]

# Wyświetlanie powtarzających się par
print(powtarzajace_sie)
