czy_wiersz_przekreslony <- function(nr_wiersza) {
  # Sprawdzenie, czy wiersz istnieje
  if (nr_wiersza <= length(arkusz$sheet_data$rows)) {
    wiersz <- arkusz$sheet_data$rows[[nr_wiersza]]
    
    # Iterowanie przez komórki w wierszu, jeśli istnieją
    if (!is.null(wiersz$c)) {
      for (nr_kolumny in seq_along(wiersz$c)) {
        komorka <- wiersz$c[[nr_kolumny]]
        nr_stylu <- komorka$s
        if (!is.null(nr_stylu) && !is.null(style_komorek[[as.numeric(nr_stylu)]]$font$strike)) {
          if (style_komorek[[as.numeric(nr_stylu)]]$font$strike == TRUE) {
            return(TRUE)  # Wiersz ma przekreśloną komórkę
          }
        }
      }
    }
  }
  return(FALSE)  # Wiersz nie ma przekreślonej komórki
}
