install.packages("readxl")
library(readxl)
# Określ ścieżkę do pliku Excel
file_path <- "ścieżka/do/pliku.xlsx"

# Określ nazwę arkusza
sheet_name <- "nazwa_arkusza"

# Wczytaj dane z określonego zakresu wierszy i kolumn
data <- read_excel(file_path, sheet = sheet_name, range = cell_rows(1:10), range = cell_cols(2:5))

# Wyświetl wczytane dane
print(data)
