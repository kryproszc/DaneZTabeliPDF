import pandas as pd

# Przykładowy DataFrame z datami oddzielonymi kropkami
data = {
    'Data początku obowiązywania okresu ubezpieczenia': ['01.01.2009', '15.05.2010', '20.07.2015'],
    'Data końca obowiązywania okresu ubezpieczenia': ['31.12.2023', '01.01.2024', '30.06.2025']
}
DaneObiekty = pd.DataFrame(data)

# Konwersja kolumn do daty z uwzględnieniem formatu dd.mm.yyyy
DaneObiekty['Data początku obowiązywania okresu ubezpieczenia'] = pd.to_datetime(DaneObiekty['Data początku obowiązywania okresu ubezpieczenia'], format='%d.%m.%Y')
DaneObiekty['Data końca obowiązywania okresu ubezpieczenia'] = pd.to_datetime(DaneObiekty['Data końca obowiązywania okresu ubezpieczenia'], format='%d.%m.%Y')

# Warunki testowe
rok_2010 = pd.Timestamp('2010-01-01')
data_2023 = pd.Timestamp('2023-12-31')

# Sprawdzenie warunków
mask_start_date = DaneObiekty['Data początku obowiązywania okresu ubezpieczenia'] >= rok_2010
mask_end_date = DaneObiekty['Data końca obowiązywania okresu ubezpieczenia'] > data_2023

# Łączenie warunków
final_mask = mask_start_date & mask_end_date

print(final_mask)

# Wyświetlanie wyników
print("Warunki spełnione dla wierszy:")
print(DaneObiekty[final_mask])

print("Warunki niespełnione dla wierszy:")
print(DaneObiekty[~final_mask])
