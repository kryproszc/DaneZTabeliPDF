# Zamiana liczb na 1, NaN pozostają bez zmian

def create_pattern_matrix(reduced_df, pattern_number):
    size = reduced_df.shape[0]-1
    reduced_df_copy = reduced_df.copy()
    for j in range(size):
        for i in range(size-pattern_number-j):
            reduced_df_copy.iloc[i,j] = 0
    return reduced_df_copy
