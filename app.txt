# Usunięcie duplikatów pozostawiając jedno przypisanie dla każdego punktu
coordinates_with_gminy_unique <- coordinates_with_gminy %>%
  group_by(geometry) %>%
  filter(row_number() == 1) %>%
  ungroup()


budynki_w_gminach <- coordinates_with_gminy_unique %>%
  group_by(REGON) %>%
  summarise(total_budynkow = sum(ilosc, na.rm = TRUE))

punkty_gminy_z_budynkami <- left_join(punkty_gminy, budynki_w_gminach, by = "REGON")
