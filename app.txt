library(shiny)
library(DT)  # Pakiet do wy≈õwietlania interaktywnych tabel

ui <- fluidPage(
  titlePanel("Wczytywanie danych CSV"),
  sidebarLayout(
    sidebarPanel(
      fileInput("file", "Wybierz plik CSV", accept = ".csv")
    ),
    mainPanel(
      textOutput("message"),
      DTOutput("table")
    )
  )
)

server <- function(input, output, session) {
  
  data <- reactive({
    req(input$file)
    df <- read.csv(input$file$datapath)
    return(df)
  })
  
  output$message <- renderText({
    if (is.null(input$file)) {
      return("Czekam na dane...")
    } else {
      return(NULL)
    }
  })
  
  output$table <- renderDT({
    data()
  })
}

shinyApp(ui, server)
