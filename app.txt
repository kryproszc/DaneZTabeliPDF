def triangle_forward(self, df_data, f):
    mm, nn = df_data.shape[0], df_data.shape[1]
    df_t_copy = df_data.copy()
    
    if len(f) > mm:
        for k in range(mm + 1, len(f) + 2):
            df_t_copy[k] = np.nan
    
    for j in range(len(f)):
        max_ind_row = np.max([0, mm - j - 1])
        for i in range(max_ind_row):
            df_t_copy.iloc[i, j + 1] = df_t_copy.iloc[i, j] * f[j]
    
    return df_t_copy
