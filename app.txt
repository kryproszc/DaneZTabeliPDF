# Załaduj potrzebne biblioteki
library(ggplot2)
library(reshape2)

# Przykład dla szkód osobowych

# Przekształcenie dataframe na format długi
# W tym przykładzie df_osobowe to Twoja macierz szkód osobowych
# Zamiast "Rok_poczatkowy" będziemy używać indeksów wierszy jako identyfikatorów
df_osobowe_long <- melt(df_osobowe, id.vars = rownames(df_osobowe))

# Ponieważ Twoje kolumny są nazwane jako liczby (np. 3, 15, 27), upewnij się, że są traktowane jako zmienne kategoryczne
df_osobowe_long$variable <- as.factor(df_osobowe_long$variable)

# Tworzenie wykresu dla szkód osobowych
ggplot(df_osobowe_long, aes(x = variable, y = value, group = Var1, color = as.factor(Var1))) +
  geom_line() +
  labs(x = "Okres rozwoju (miesiące)", y = "Wartość szkód", title = "Rozwój szkód osobowych w czasie") +
  theme_minimal()
