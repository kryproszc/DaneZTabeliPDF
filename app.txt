library(dplyr)

# Krok 1: Filtrowanie wierszy, gdzie druga kolumna zawiera "PZU SA"
wynik <- df_st_sum6 %>% filter(`...3` == "PZU SA")

# Krok 2: Przekształcenie odpowiednich kolumn na typ numeryczny
# Zakładam, że chcesz, aby wszystkie kolumny oprócz pierwszej i drugiej były numeryczne
# Używamy mutate_at, aby przekształcić wybrane kolumny

# Poniższy kod zakłada, że od 4 kolumny wzwyż mają być liczby
# (zamiast tego możesz wybrać specyficzne kolumny np. 4:10)
wynik <- wynik %>% mutate_at(vars(-V1, -`...3`), as.numeric)

# Krok 3: Wyświetlenie wyniku
print(wynik)
