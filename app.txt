library(dplyr)

# Zakładam, że df_wspolrzedne jest ramką danych z kolumnami Szerokosc i Dlugosc

# Zliczanie liczby powtórzeń dla każdej kombinacji Szerokości i Długości
wystapienia <- df_wspolrzedne %>%
  group_by(Szerokosc, Dlugosc) %>%
  summarise(Freq = n()) %>%
  filter(Freq > 1)  # Filtrujemy tylko te kombinacje, które się powtarzają

# Łączenie z oryginalnym dataframe, aby wyświetlić powtarzające się wiersze
duplikaty <- inner_join(df_wspolrzedne, wystapienia, by = c("Szerokosc", "Dlugosc"))

# Wyświetlenie unikalnych powtarzających się wierszy z liczbą wystąpień
duplikaty_unikalne <- unique(duplikaty[, c("Szerokosc", "Dlugosc", "Freq")])

# Wyświetlenie wyników
print(duplikaty_unikalne)
