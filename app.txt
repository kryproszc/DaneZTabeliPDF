czy_jest_ciag <- function(wektor) {
  for (i in 1:(length(wektor) - 1)) {
    if (wektor[i + 1] == wektor[i] + 1) {
      return(TRUE)  # Zwraca TRUE, jeśli znaleziono ciąg
    }
  }
  return(FALSE)  # Zwraca FALSE, jeśli nie znaleziono ciągu
}

# Przykładowe użycie
wektor <- c(3, 5, 4, 6, 7)
wynik <- czy_jest_ciag(wektor)
print(wynik)
