# Funkcja do konwersji dat z uwzględnieniem różnych formatów
def convert_date(date_str):
    try:
        return pd.to_datetime(date_str, format='%Y.%m.%d')
    except ValueError:
        return pd.to_datetime(date_str, format='%Y-%m-%d')

# Zastosowanie funkcji do konwersji kolumn
DaneObiekty['Data początku obowiązywania okresu ubezpieczenia'] = DaneObiekty['Data początku obowiązywania okresu ubezpieczenia'].apply(convert_date)
DaneObiekty['Data końca obowiązywania okresu ubezpieczenia'] = DaneObiekty['Data końca obowiązywania okresu ubezpieczenia'].apply(convert_date)

# Warunki testowe
rok_2010 = pd.Timestamp('2010-01-01')
data_2023 = pd.Timestamp('2023-12-31')

# Sprawdzenie warunków
mask_start_date = DaneObiekty['Data początku obowiązywania okresu ubezpieczenia'] >= rok_2010
mask_end_date = DaneObiekty['Data końca obowiązywania okresu ubezpieczenia'] > data_2023

# Łączenie warunków
final_mask = mask_start_date & mask_end_date
