# Przypisanie najbliższej gminy do każdego punktu za pomocą st_nearest_feature
# Znalezienie indeksu najbliższej gminy dla każdego punktu
nearest_gminy_indices <- st_nearest_feature(coordinates_sf, punkty_gminy)

# Dodanie informacji o gminach do punktów
coordinates_with_gminy <- cbind(coordinates_sf, punkty_gminy[nearest_gminy_indices, ])
