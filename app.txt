# app.R

library(shiny)

ui <- fluidPage(
  titlePanel("Miejsca zerowe funkcji kwadratowej"),
  
  sidebarLayout(
    sidebarPanel(
      numericInput("a", "Podaj współczynnik a:", value = 1),
      numericInput("b", "Podaj współczynnik b:", value = 0),
      numericInput("c", "Podaj współczynnik c:", value = 0),
      actionButton("calculate", "Oblicz miejsca zerowe")
    ),
    mainPanel(
      verbatimTextOutput("result")
    )
  )
)

server <- function(input, output) {
  observeEvent(input$calculate, {
    a <- input$a
    b <- input$b
    c <- input$c
    
    delta <- b^2 - 4*a*c
    if (delta < 0) {
      result <- "Funkcja nie ma miejsc zerowych."
    } else if (delta == 0) {
      result <- paste("Miejsce zerowe funkcji:", -b / (2*a))
    } else {
      root1 <- (-b + sqrt(delta)) / (2*a)
      root2 <- (-b - sqrt(delta)) / (2*a)
      result <- paste("Miejsca zerowe funkcji:", root1, root2)
    }
    
    output$result <- renderText({
      result
    })
  })
}

shinyApp(ui = ui, server = server)
