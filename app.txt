znajdz_przerwy <- function(wektor) {
  przerwy <- c()  # Zmienna do przechowywania elementów, które są przerwami
  for (i in 1:(length(wektor) - 1)) {
    if (wektor[i + 1] - wektor[i] > 1) {
      przerwy <- c(przerwy, wektor[i], wektor[i + 1])  # Dodajemy elementy z przerwą
    }
  }
  return(przerwy)  # Zwracamy znalezione elementy z przerwami
}
