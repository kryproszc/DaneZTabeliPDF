kody <- trimws(kody)

# Usuwanie spacji w środku kodu, np. "22 -111" na "22-111"
kody <- gsub("\\s+", "", kody)

# Usunięcie wartości NA i pustych miejsc
kody <- kody[!is.na(kody) & kody != ""]

# 2. Sprawdzenie poprawności formatu XX-XXX
# Definiujemy wzorzec dla kodu pocztowego w Polsce
valid_pattern <- "^\\d{2}-\\d{3}$"

# Tworzenie logicznego wektora z informacją, które kody są poprawne
valid_kody <- grepl(valid_pattern, kody)

# Filtracja: pozostawiamy tylko kody, które mają poprawny format
kody_poprawne <- kody[valid_kody]

# 3. Wynik: wyświetlenie poprawnych kodów pocztowych
print("Poprawne kody pocztowe:")
print(kody_poprawne)
