# Funkcja do znajdowania komórek zawierających problematyczne znaki w nazwach
find_problematic_cells_in_names <- function(df, column_name, pattern) {
  # Znajdź wiersze, które zawierają problematyczne znaki
  problematic_rows <- which(grepl(pattern, df[[column_name]]))
  
  # Zwróć wiersze i komórki, które mają problematyczne znaki
  if (length(problematic_rows) > 0) {
    result <- data.frame(Row = problematic_rows, 
                         Value = df[[column_name]][problematic_rows],
                         stringsAsFactors = FALSE)
    return(result)
  } else {
    return("Brak problematycznych znaków w tej kolumnie.")
  }
}
