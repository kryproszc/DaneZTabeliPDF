library(dplyr)
library(openxlsx)

# Ścieżka do pliku Excel
file_path <- "path_to_your_excel_file.xlsx"

# Wczytanie całego arkusza
sheet_data <- read.xlsx(file_path, sheet = "Sheet1")

# Konwersja kolumn na liczby typu float
numeric_data <- sheet_data %>%
  mutate(across(everything(), ~ as.numeric(gsub(",", ".", .))))

# Wyświetlenie przekonwertowanego DataFrame
print(numeric_data)
