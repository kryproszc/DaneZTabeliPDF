triangle_forward <- function(triangle, f) {
  mm <- nrow(triangle)
  nn <- ncol(triangle)
  
  df_t_copy <- triangle
  
  if (length(f) > mm) {
    for (k in (mm + 1):(length(f) + 2)) {
      df_t_copy[k, ] <- NA
    }
  }
  
  for (j in 1:length(f)) {
    max_ind_row <- max(0, mm - j)
    for (i in 1:max_ind_row) {
      df_t_copy[i, j + 1] <- df_t_copy[i, j] * f[j]
    }
  }
  
  return(df_t_copy)
}
