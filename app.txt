combined_pattern <- paste(problematic_patterns, collapse = "|")

# Funkcja do zamiany problematycznych znaków na puste miejsca tylko w kolumnach tekstowych
replace_problematic_chars <- function(df, pattern) {
  # Sprawdzanie tylko kolumn tekstowych (charakterowych)
  is_char_col <- sapply(df, is.character)
  
  # Zamiana problematycznych znaków w kolumnach tekstowych
  df[, is_char_col] <- lapply(df[, is_char_col], function(col) gsub(pattern, "", col))
  
  return(df)
}

# Zamiana problematycznych znaków na puste miejsca w kolumnach tekstowych
data_input <- replace_problematic_chars(data_input, combined_pattern)
