data <- data.frame(
  osobowe = as.vector(osobowe_increments),
  majatkowe = as.vector(majatkowe_increments),
  paid6 = as.vector(majatkowe_osobowe_increments)
)

# Przekształcamy dane na format długi (long format), aby móc łatwo użyć facetowania
data_long <- reshape2::melt(data, id.vars = NULL)

# Tworzymy wykresy rozrzutu dla wszystkich trzech zależności obok siebie
ggplot(data_long, aes(x = value, y = paid6, color = variable)) +
  geom_point() +
  facet_wrap(~variable, scales = "free", labeller = labeller(variable = c("osobowe" = "Szkody osobowe vs Paid (6)", 
                                                                          "majatkowe" = "Szkody majątkowe vs Paid (6)", 
                                                                          "paid6" = "Paid (6)"))) +
  labs(title = "Wykresy rozrzutu dla różnych typów szkód", 
       x = "Inkrementy szkód", 
       y = "Paid (6) (Inkrementy)") +
  theme_minimal() +
  theme(legend.position = "none")
