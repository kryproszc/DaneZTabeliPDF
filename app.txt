# Załaduj potrzebne biblioteki
library(ggplot2)
library(reshape2)

# Przykład dla szkód osobowych

# Załóżmy, że Twoje dane nazywają się df_osobowe

# Przekształcenie indeksów w nową kolumnę
df_osobowe$Rok_poczatkowy <- rownames(df_osobowe)

# Przekształcenie dataframe na format długi
df_osobowe_long <- melt(df_osobowe, id.vars = "Rok_poczatkowy")

# Upewnienie się, że zmienne (kolumny) są traktowane jako zmienne kategoryczne
df_osobowe_long$variable <- as.factor(df_osobowe_long$variable)

# Tworzenie wykresu dla szkód osobowych
ggplot(df_osobowe_long, aes(x = variable, y = value, group = Rok_poczatkowy, color = as.factor(Rok_poczatkowy))) +
  geom_line() +
  labs(x = "Okres rozwoju (miesiące)", y = "Wartość szkód", title = "Rozwój szkód osobowych w czasie") +
  theme_minimal()
