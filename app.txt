// JavaScript for handling clicks in Paid Claims
$(document).on('click', '#ratios-table1 td', function() {
    var row = $(this).closest('tr').index();
    var col = $(this).index();
    console.log('Clicked cell (Paid Claims): row ' + row + ', col ' + col); // Logowanie
    if ($(this).hasClass('highlighted')) {
        $(this).removeClass('highlighted');
        Shiny.setInputValue('panel1_clicked_cell', {row: row, col: col, highlighted: false});
    } else {
        $(this).addClass('highlighted');
        Shiny.setInputValue('panel1_clicked_cell', {row: row, col: col, highlighted: true});
    }
});
