data_input <- read.csv("sciezka/do/pliku.csv", 
                       sep = ";",        # Średnik jako separator kolumn
                       dec = ",",        # Przecinek jako separator dziesiętny
                       fileEncoding = "UTF-8-BOM",  # Kodowanie pliku
                       header = FALSE,   # Brak nagłówków (wczytaj jako dane)
                       stringsAsFactors = FALSE,  # Unikaj zamiany tekstu na faktory
                       check.names = FALSE)  # Pozostaw oryginalne nazwy kolumn bez modyfikacji

# Ustawienie pierwszego wiersza jako nazwy kolumn
colnames(data_input) <- as.character(unlist(data_input[1, ]))

# Usunięcie pierwszego wiersza (ponieważ to teraz nagłówki)
data_input <- data_input[-1, ]

# Usunięcie pustych kolumn (jeśli jakieś istnieją, np. V18)
data_input <- data_input[, colSums(!is.na(data_input)) > 0]
