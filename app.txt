import pandas as pd
import re

# Przykładowy DataFrame
data = {
    'Adres ubezpieczonego obiektu - Miasto': ['Warsaw', 'Krakow 2', 'Poznan', 'Lodz1', 'Gdansk', 'Wroclaw3']
}
DaneObiekty = pd.DataFrame(data)

# Wyrażenie regularne dla sprawdzenia liczb w stringu
number_pattern = re.compile(r'\d')

# Funkcja sprawdzająca, czy wartość zawiera liczby
def contains_number(value):
    return bool(number_pattern.search(value))

# Zastosowanie funkcji do kolumny
mask_contains_number = DaneObiekty['Adres ubezpieczonego obiektu - Miasto'].apply(contains_number)

# Wyświetlanie wyników
print(mask_contains_number)

print("Wiersze z liczbami w kolumnie 'Adres ubezpieczonego obiektu - Miasto':")
print(DaneObiekty[mask_contains_number])

print("Wiersze bez liczb w kolumnie 'Adres ubezpieczonego obiektu - Miasto':")
print(DaneObiekty[~mask_contains_number])
