import requests

# adres do geokodowania
adres = "1600 Pennsylvania Ave NW, Washington, DC"

# tworzenie żądania do Nominatim API
url = f"https://nominatim.openstreetmap.org/search?format=json&q={adres}"

# wysłanie żądania
response = requests.get(url, headers={"User-Agent": "Mozilla/5.0"})

# przetwarzanie odpowiedzi
if response.status_code == 200:
    wyniki = response.json()
    if wyniki:
        lokalizacja = wyniki[0]
        print("Współrzędne:", lokalizacja['lat'], lokalizacja['lon'])
    else:
        print("Brak wyników dla podanego adresu.")
else:
    print("Nie udało się przetworzyć żądania.")
