void create_custom_directory(const fs::path &path)
{
    if (!fs::exists(path))
    {
        if (fs::create_directories(path))
        {
            std::cout << "Utworzono folder: " << path << std::endl;
        }
        else
        {
            std::cout << "Błąd: " << path << std::endl;
        }
    }
}
